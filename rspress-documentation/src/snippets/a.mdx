import { Elysia /*, file */ } from 'elysia'

new Elysia()
  .get('/', 'Hello A')
  // .get('/image', file('mika.webp'))
  .get('/stream', function* () {
    yield 'Hello'
    yield 'World'
  })
  .ws('/realtime', {
    message(ws, message) {
      ws.send('got:' + message)
    },
  })
  .listen(3000)
